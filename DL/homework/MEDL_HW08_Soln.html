Problem 1: MNIST Digit Classification with Insufficient Data¶

In many real problems, lack of data is a critical issue. Especially, engineering problems rarely have fault data since they have to be collected when the system is abnoramlly operating. Thus, it is essential to train the model with a small data set. One possible method is the transfer learning that we learned in class.

In this problem, we are going to classify MNIST digit images with insufficient train data. First, we will train a CNN model with a small dataset. And then, we will train the same model but using transfer learning to compare both models.

For transfer learning, we will use the EMNIST alphabet dataset.

# input layer
input_h = 28 # Input height
input_w = 28 # Input width
input_ch = 1 # Input channel : Gray scale
# (None, 28, 28, 1)

# First convolution layer
k1_h = 3
k1_w = 3
k1_ch = 32 
p1_h = 2
p1_w = 2
# (None, 14, 14 ,32)

# Second convolution layer
k2_h = 3
k2_w = 3
k2_ch = 64
p2_h = 2
p2_w = 2
# (None, 7, 7 ,64)

## Fully connected: flatten the features -> (None, 7*7*64)
conv_result_size = int(input_h/(p1_h*p2_h)) * int(input_w/(p1_w*p2_w)) * k2_ch
n_hidden = 100

# kernel size: [kernel_height, kernel_width, input_ch, output_ch]
weights = {
    'conv1' : tf.Variable(tf.random_normal([k1_h, k1_w, input_ch, k1_ch], stddev = 0.1)),
    'conv2' : tf.Variable(tf.random_normal([k2_h, k2_w, k1_ch, k2_ch], stddev = 0.1)),
    'hidden' : tf.Variable(tf.random_normal([conv_result_size, n_hidden], stddev = 0.1)),
    'output_10_1' : tf.Variable(tf.random_normal([n_hidden, 10], stddev = 0.1)),
    'output_10_2' : tf.Variable(tf.random_normal([n_hidden, 10], stddev = 0.1)),
    'output_26' : tf.Variable(tf.random_normal([n_hidden, 26], stddev = 0.1))
}

# bias size: [output_ch] or [neuron_size]
biases = {
    'conv1' : tf.Variable(tf.random_normal([k1_ch], stddev = 0.1)),
    'conv2' : tf.Variable(tf.random_normal([k2_ch], stddev = 0.1)),
    'hidden' : tf.Variable(tf.random_normal([n_hidden], stddev = 0.1)),
    'output_10_1' : tf.Variable(tf.random_normal([10], stddev = 0.1)),
    'output_10_2' : tf.Variable(tf.random_normal([10], stddev = 0.1)),
    'output_26' : tf.Variable(tf.random_normal([26], stddev = 0.1))
}

# input layer: [batch_size, image_height, image_width, channels]
# output layer: [batch_size, class_size]
x = tf.placeholder(tf.float32, [None, input_h, input_w, input_ch])
y_26 = tf.placeholder(tf.float32, [None, 26])
y_10 = tf.placeholder(tf.float32, [None, 10])

# [batch, height, width, channels]

# First convolution layer
conv1 = tf.nn.conv2d(x, 
                     weights['conv1'], 
                     strides = [1, 1, 1, 1], 
                     padding = 'SAME')
conv1 = tf.nn.relu(tf.add(conv1, biases['conv1']))
maxp1 = tf.nn.max_pool(conv1, 
                       ksize = [1, p1_h, p1_w, 1], 
                       strides = [1, p1_h, p1_w, 1], 
                       padding = 'VALID')

# Second convolution layer
conv2 = tf.nn.conv2d(maxp1, 
                     weights['conv2'], 
                     strides = [1, 1, 1, 1], 
                     padding = 'SAME')
conv2 = tf.nn.relu(tf.add(conv2, biases['conv2']))
maxp2 = tf.nn.max_pool(conv2,
                       ksize = [1, p2_h, p2_w, 1], 
                       strides = [1, p2_h, p2_w, 1], 
                       padding = 'VALID')

maxp2_flatten = tf.reshape(maxp2, [-1, conv_result_size])

# Fully connected
hidden = tf.add(tf.matmul(maxp2_flatten, weights['hidden']), biases['hidden'])
hidden = tf.nn.relu(hidden)
pred_10_1 = tf.add(tf.matmul(hidden, weights['output_10_1']), biases['output_10_1'])
pred_10_2 = tf.add(tf.matmul(hidden, weights['output_10_2']), biases['output_10_2'])
pred_26 = tf.add(tf.matmul(hidden, weights['output_26']), biases['output_26'])

LR = 0.001

loss_10_1 = tf.reduce_mean(tf.nn.softmax_cross_entropy_with_logits_v2(labels = y_10, logits = pred_10_1))
loss_10_2 = tf.reduce_mean(tf.nn.softmax_cross_entropy_with_logits_v2(labels = y_10, logits = pred_10_2))
loss_26 = tf.reduce_mean(tf.nn.softmax_cross_entropy_with_logits_v2(labels = y_26, logits = pred_26))

optm_10_1 = tf.train.AdamOptimizer(LR).minimize(loss_10_1)
optm_10_2 = tf.train.AdamOptimizer(LR).minimize(loss_10_2, var_list=[weights['output_10_2'], biases['output_10_2']])
optm_26 = tf.train.AdamOptimizer(LR).minimize(loss_26)

def next_batch(x, y, batch_size):
    random_idx = np.random.randint(len(y), size = batch_size)
    return x[random_idx], y[random_idx]

n_batch = 50
n_iter = 250
n_prt = n_iter//20

sess = tf.Session()
init = tf.global_variables_initializer()
sess.run(init)

loss_record_train = []
loss_record_test = []
for epoch in range(n_iter+1):
    train_x, train_y = mnist.train.next_batch(n_batch)
    train_x = np.reshape(train_x, [-1, input_h, input_w, input_ch])
    sess.run(optm_10_1, feed_dict = {x: train_x,  y_10: train_y})
    
    if epoch % n_prt == 0:
        test_x, test_y = mnist.test.next_batch(n_batch)
        test_x = np.reshape(test_x, [-1, input_h, input_w, input_ch])
        c1 = sess.run(loss_10_1, feed_dict = {x: train_x, y_10: train_y})
        c2 = sess.run(loss_10_1, feed_dict = {x: test_x, y_10: test_y})
        loss_record_train.append(c1)
        loss_record_test.append(c2)
        print ("({:04d}/{:04d}) loss_train: {:.6f}, loss_test: {:.6f}".format(epoch, n_iter, c1, c2))

plt.plot(np.arange(len(loss_record_train))*n_prt, loss_record_train, label = 'train')
plt.plot(np.arange(len(loss_record_test))*n_prt, loss_record_test, label = 'test')
plt.xlabel('iteration', fontsize = 15)
plt.ylabel('loss', fontsize = 15)
plt.legend(fontsize = 12)
plt.ylim([0, np.max(loss_record_train)])

def net(x, weights, biases):
    conv1 = tf.nn.conv2d(x, weights['conv1'], strides= [1, 1, 1, 1], padding = 'SAME')
    conv1 = tf.nn.relu(tf.add(conv1, biases['conv1']))
    maxp1 = tf.nn.max_pool(conv1, ksize = [1, p1_h, p1_w, 1], strides = [1, p1_h, p1_w, 1], padding = 'VALID', name = 'maxp1')
    
    conv2 = tf.nn.conv2d(maxp1, weights['conv2'], strides= [1, 1, 1, 1], padding = 'SAME')
    conv2 = tf.nn.relu(tf.add(conv2, biases['conv2']))
    maxp2 = tf.nn.max_pool(conv2, ksize = [1, p2_h, p2_w, 1], strides = [1, p2_h, p2_w, 1], padding = 'VALID', name = 'maxp1')
    
    maxp2_re = tf.reshape(maxp2, [-1, conv_result_size])
    
    hidden = tf.add(tf.matmul(maxp2_re, weights['hidden']), biases['hidden'])
    hidden = tf.nn.relu(hidden)
    
    score = tf.add(tf.matmul(hidden, weights['output']), biases['output'], name= 'score')

    return score

def encoder(x):        
    ## First convolution layer
    conv1 = tf.layers.conv2d(inputs = x, 
                             filters = 32, 
                             kernel_size = [3, 3], 
                             strides = [1, 1],
                             padding = "SAME", 
                             activation = tf.nn.relu)
    maxp1 = tf.layers.max_pooling2d(inputs = conv1, 
                                    pool_size = [2, 2], 
                                    strides = 2)

    ## Second convolution layer
    conv2 = tf.layers.conv2d(inputs = maxp1, 
                             filters = 64, 
                             kernel_size = [3, 3], 
                             padding = "SAME", 
                             activation = tf.nn.relu)
    maxp2 = tf.layers.max_pooling2d(inputs = conv2, 
                                    pool_size = [2, 2], 
                                    strides = 2)

    conv3 = tf.layers.conv2d(inputs = maxp2, 
                             filters = 10, 
                             kernel_size = [25, 25], 
                             padding = "VALID")

    return conv3

def decoder(latent):        
    deconv3 = tf.layers.conv2d_transpose(inputs = latent, 
                                         filters = 64, 
                                         kernel_size = [25, 25], 
                                         padding = 'VALID',
                                         activation = tf.nn.relu)

    ## First deconvolution layer
    deconv2 = tf.layers.conv2d_transpose(inputs = deconv3, 
                                         filters = 32, 
                                         kernel_size = [3, 3], 
                                         strides = (2, 2),
                                         padding = 'SAME',
                                         activation = tf.nn.relu)

    ## Second deconvolution layer    
    deconv1 = tf.layers.conv2d_transpose(inputs = deconv2, 
                                         filters = 1, 
                                         kernel_size = [3, 3], 
                                         strides = (2, 2),
                                         padding = 'SAME')

    return deconv1
